### 堆的定义

  *  堆是一个完全二叉树；
  *  堆中每一个节点的值都必须大于等于（或小于等于）其子树中每个节点的值。


```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findKthLargest = function (nums, k) {
  const buildHeap = (a) => {
    const n = a.length;
    for (let i = n >> 1; i >= 0; --i) {
      heapify(a, n, i);
    }
    return a;
  };
  const heapify = (a, n, i) => {
    while (true) {
      let maxPos = i;
      if (i * 2 + 1 < n && a[i] < a[i * 2 + 1]) maxPos = i * 2 + 1;
      if (i * 2 + 2 < n && a[maxPos] < a[i * 2 + 2]) maxPos = i * 2 + 2;
      if (maxPos == i) break;
      swap(a, i, maxPos);
      i = maxPos;
    }
  };
  const swap = (a, i, max) => {
    let temp = a[i];
    a[i] = a[max];
    a[max] = temp;
  };
  const arr = buildHeap(nums);
  let v;
  while (k > 0) {
    swap(arr, 0, arr.length - 1);
    v = arr.pop();
    heapify(arr, arr.length, 0);
    k--;
  }
  return v;
};
```
